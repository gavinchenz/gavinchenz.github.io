function routesData(result){
    result.forEach(item=>{
        if(item.component === 'layout'){
            routes.push({
                path:item.path,
                //name:item.name,
                redirect:item.redirect,
                component:()=> import(`@/Layout`),
                children:getChildren(item.children)
            })
        }else{
            routes.push({
                path:item.path,
                //name:item.name,
                redirect:item.redirect,
                component:()=> import(`@/views/${item.component}`),
                children:getChildren(item.children)
            })
        }
    })
    return routes;
}

//å¤„ç†å­è·¯ç”±
function getChildren(result){
    if(result && result.length){
        const children = [];
        result.forEach(item=>{
            children.push({
                path:item.path,
                //name:item.name,
                redirect:item.redirect || '',
                component:()=> import(`@/views/${item.component}`),
                children:getChildren(item.children)
            })
        })
        return children;
    }
    return [];
}
